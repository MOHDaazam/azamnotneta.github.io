  let cache = {};
  return function(x, y) {
    const key = `${x}-${y}`;
    if (key in cache) {
      return cache[key];
    } else {
      const result = x + y;
      cache[key] = result;
      return result;
    }
  };
}

console.log(memoizedAddition(2, 3)); // Output: 5 (from cache)
